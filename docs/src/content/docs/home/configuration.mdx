import { RyanlinkManager, QueueChangesWatcher, QueueStoreManager } from 'ryanlink'
import { RedisClientType } from 'redis' // example for custom queue store
import { Client, GatewayIntentBits } from 'discord.js' // example for a discord bot

// you might want to extend the types of the client, to bind ryanlink to it.
const client = new Client({
  intents: [GatewayIntentBits.Guilds, GatewayIntentBits.GuildVoiceStates],
})

const previouslyUsedSessions = new Map<string, string>() //nodeManager.on("connect", node => previouslyUsedSessions.set(node.id, node.sessionId))

client.ryanlink = new RyanlinkManager({
  nodes: [
    {
      authorization: 'localhoist',
      host: 'localhost',
      port: 2333,
      id: 'testnode',
      
      // Resuming support
      sessionId: previouslyUsedSessions.get('testnode'), // session ID from a previous connection
      
      // Connection Stability
      requestSignalTimeoutMS: 3000, // Timeout for REST requests to the node.
      heartBeatInterval: 30_000, // Frequency of keep-alive packets.
      closeOnError: true, // Whether to close the connection if a critical error occurs.
      
      // Performance & Stats
      enablePingOnStatsCheck: true, // Perform a ping check whenever stats are received.
      retryAmount: 5, // Number of reconnection attempts before giving up.
      retryDelay: 10e3, // Delay (in ms) between reconnection attempts.
      
      // Security & Validation
      secure: false, // Use wss:// and https:// if the node requires SSL.
      autoChecks: {
        pluginValidations: true, // Validate whether the node supports required plugins on startup.
        sourcesValidations: true, // Validate whether the node supports required audio sources.
      },
    },
  ],
  sendToShard: (guildId, payload) => client.guilds.cache.get(guildId)?.shard?.send(payload),
  
  // High-level Automation
  autoSkip: true, // Automatically skip to the next track on track-end, error, or stuck.
  autoMove: false, // Auto-move a player to a new node if its current node disconnects.
  autoSkipOnResolveError: true, // Skip the current song if an UnresolvedTrack fails to resolve properly.
  emitNewSongsOnly: true, // Only emit 'trackStart' for new songs (prevents spam when looping).
  
  // Whitelist / Blacklist for safety
  linksAllowed: true, // Whether to allow raw URL playback.
  linksBlacklist: [], // Domains or keywords to block from playback (Regex supported).
  linksWhitelist: [], // Domains or keywords that are always allowed.

  client: {
    id: envConfig.clientId, // REQUIRED for signaling and voice state management.
    username: 'RyanlinkBot',
  },

  playerOptions: {
    // Stability thresholds
    maxErrorsPerTime: {
      threshold: 10_000,
      maxAmount: 3, // Destroy player if it hits more than 3 errors in 10 seconds.
    },
    
    // Autoplay configuration
    minAutoPlayMs: 10_000, // Cooldown for autoplay triggers to prevent spam.
    
    // Audio manipulation
    applyVolumeAsFilter: false, // Whether to apply volume via a filter for smoother transitions.
    volumeDecrementer: 0.75, // Scale client volume (e.g., 100% in client = 75% on Lavalink).
    
    // Advanced player state
    clientBasedPositionUpdateInterval: 50, // Frequency (in ms) to locally predict track position.
    defaultSearchPlatform: 'ytmsearch', // Default prefix for queries without a source.
    allowCustomSources: false, // Allow playback of URLs not natively supported by Lavalink plugins.
    useUnresolvedData: true, // Optimize memory by using unresolved metadata instead of full track data.
    
    requesterTransformer: requesterTransformer, // Function to normalize requester objects.
    
    onDisconnect: {
      autoReconnect: true, // Attempt to reconnect the player if the bot is kicked from VC.
      destroyPlayer: false, // Instantly destroy the player on any voice disconnect (overrides autoReconnect).
    },
    
    onEmptyQueue: {
      destroyAfterMs: 30_000, // Duration to wait before auto-destroying an empty player.
      autoPlayFunction: autoPlayFunction, // Handler to fetch more tracks when the queue is dry.
    },
  },

  queueOptions: {
    maxPreviousTracks: 10, // How many played tracks to keep in history.
    queueStore: new myCustomStore(client.redis), // Custom manager for external queue persistence (Redis/DB).
    queueChangesWatcher: new myCustomWatcher(client), // Callback listener for queue mutations.
  },

  advancedOptions: {
    enableDebugEvents: true, // Required to listen to the 'debug' event.

      logCustomSearches: false, // Debug output for raw search plugin query handling.
    },
  },
})
// IMPORTANT- Send voice data to the audio node
client.on('raw', (d) => client.ryanlink.provideVoiceUpdate(d))

client.on('ready', () => {
  client.ryanlink.init(client.user) // init ryanlink
})

// for the custom queue Store create a redis instance
client.redis = createClient({ url: 'redis://localhost:6379', password: 'securepass' })
client.redis.connect()

// Custom external queue Store
// if queueStore is not provided on default it uses MiniMap
export class myCustomStore implements QueueStoreManager {
  private redis: RedisClientType
  constructor(redisClient: RedisClientType) {
    this.redis = redisClient
  }
  async get(guildId): Promise<any> {
    return await this.redis.get(this.id(guildId))
  }
  async set(guildId, stringifiedQueueData): Promise<any> {
    return await this.redis.set(this.id(guildId), stringifiedQueueData)
  }
  async delete(guildId): Promise<any> {
    return await this.redis.del(this.id(guildId))
  }
  async parse(stringifiedQueueData): Promise<Partial<StoredQueue>> {
    return JSON.parse(stringifiedQueueData)
  }
  async stringify(parsedQueueData): Promise<any> {
    return JSON.stringify(parsedQueueData)
  }
  private id(guildId) {
    return `ryanlink_queue_${guildId}` // transform the id to your belikings
  }
}

// Custom Queue Watcher Functions (with couple special features..)

function updateNextTrackGapLess(guildId: string) {
  // this function is for nodelink users relevant for gapless audio.
  const player = this.client.ryanlink.players.get(guildId)
  if (!player || !player.node.isNodeLink()) return
  return player.node.removeNextTrackGapLess(player)
}
class myCustomWatcher implements QueueChangesWatcher {
  constructor(client) {
    this.client = client
  }
  shuffled(guildId, oldStoredQueue, newStoredQueue) {
    console.log(`${this.client.guilds.cache.get(guildId)?.name || guildId}: Queue got shuffled`)
    updateNextTrackGapLess(guildId)
  }
  tracksAdd(guildId, tracks, position, oldStoredQueue, newStoredQueue) {
    console.log(
      `${this.client.guilds.cache.get(guildId)?.name || guildId}: ${tracks.length} Tracks got added into the Queue at position #${position}`
    )
    updateNextTrackGapLess(guildId)
  }
  tracksRemoved(guildId, tracks, position, oldStoredQueue, newStoredQueue) {
    console.log(
      `${this.client.guilds.cache.get(guildId)?.name || guildId}: ${tracks.length} Tracks got removed from the Queue at position #${position}`
    )
    updateNextTrackGapLess(guildId)
  }
}
